 * @dev –î–ª—è —Ö–µ—à–∏—Ä–æ–≤–∞–Ω–∏—è –∑–Ω–∞—á–µ–Ω–∏—è –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å –ø—É–±–ª–∏—á–Ω—É—é —Ñ—É–Ω–∫—Ü–∏—é getHash()
 */
contract CommitReveal {
    mapping (address account => bytes32 hash) private _hashes;

    event Committed(address indexed account, bytes32 hash);
    event Revealed(address indexed account, string value);

    error HashAlreadyCommitted();
    error NotCommitted();
    error ValueNotMatchHash();

    /**
     * @notice –°–æ—Ö—Ä–∞–Ω—è–µ—Ç –ø—Ä–∏–≤–∞—Ç–Ω—É—é —Ç–µ—Å—Ç–æ–≤—É—é —Å—Ç—Ä–æ–∫—É
     * @param hash –•–µ—à –ø—Ä–∏–≤–∞—Ç–Ω–æ–π —Å—Ç—Ä–æ–∫–∏
     * @dev –î–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è hash –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å –ø—É–±–ª–∏—á–Ω—É—é —Ñ—É–Ω–∫—Ü–∏—é getHash()
     */
    function commit(bytes32 hash) external {
        if (hasHash(msg.sender)) {
            // –ó–∞–ø—Ä–µ—â–∞–µ–º –ø–æ–≤—Ç–æ—Ä–Ω–æ —Å–æ—Ö—Ä–∞–Ω—è—Ç—å —Ö–µ—à
            revert HashAlreadyCommitted();
        }

        // –°–æ—Ö—Ä–∞–Ω—è–µ–º —Ö–µ—à
        _hashes[msg.sender] = hash;

        emit Committed(msg.sender, hash);
    }

    /**
     * @notice –†–∞—Å–∫—Ä—ã–≤–∞–µ—Ç –ø—Ä–∏–≤–∞—Ç–Ω—É—é —Å—Ç—Ä–æ–∫—É –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
     * @param value –†–∞—Å–∫—Ä—ã—Ç–∞—è —Å—Ç—Ä–æ–∫–∞
     */
    function reveal(string memory value) external returns (string memory) {
        if (!hasHash(msg.sender)) {
            // –û—Ç–º–µ–Ω—è–µ–º —Ç—Ä–∞–Ω–∑–∞–∫—Ü–∏—é, –µ—Å–ª–∏ —Ö–µ—à–∏—Ä–æ–≤–∞–Ω–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –Ω–µ –±—ã–ª–æ –¥–æ–±–∞–≤–ª–µ–Ω–æ
            revert NotCommitted();
        }

        // –ü–æ–ª—É—á–∞–µ–º —Ö–µ—à–∏—Ä–æ–≤–∞–Ω–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ
        bytes32 hash = getHash(msg.sender, value);

        if (hash != _hashes[msg.sender]) {
            // –û—Ç–º–µ–Ω—è–µ–º —Ç—Ä–∞–Ω–∑–∞–∫—Ü–∏—é, –µ—Å–ª–∏ —Ö–µ—à–∏—Ä–æ–≤–∞–Ω–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –Ω–µ —Å–æ–≤–ø–∞–¥–∞–µ—Ç —Å —Å–æ—Ö—Ä–∞–Ω–µ–Ω–Ω—ã–º —Ö–µ—à–∏—Ä–æ–≤–∞–Ω–Ω—ã–º –∑–Ω–∞—á–µ–Ω–∏–µ–º –Ω–∞ —Å—Ç–∞–¥–∏–∏ commit
            revert ValueNotMatchHash();
        }

        // –£–¥–∞–ª—è–µ–º —Å–æ—Ö—Ä–∞–Ω–µ–Ω–Ω–æ–µ —Ö–µ—à–∏—Ä–æ–≤–∞–Ω–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ
        // –≠—Ç–æ –ø–æ–∑–≤–æ–ª–∏—Ç –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—é –ø–æ–≤—Ç–æ—Ä–Ω–æ –≤—ã–∑–≤–∞—Ç—å commit() —Ñ—É–Ω–∫—Ü–∏—é
        delete _hashes[msg.sender];

        // –û—Ç–ø—Ä–∞–≤–ª—è–µ–º —Å–æ–±—ã—Ç–∏–µ, —Ç–µ–º —Å–∞–º—ã–º —Ä–∞—Å–∫—Ä—ã–≤–∞—è –∏ –ø–æ–¥—Ç–≤–µ—Ä–∂–¥–∞—è –ø—Ä–∞–≤–∏–ª—å–Ω–æ—Å—Ç—å —Ä–∞—Å–∫—Ä—ã—Ç–∏—è
        emit Revealed(msg.sender, value);

        return value;
    }

    /**
     * @notice –ü—Ä–æ–≤–µ—Ä—è–µ—Ç –Ω–∞–ª–∏—á–∏–µ —Å–æ—Ö—Ä–∞–Ω–µ–Ω–Ω–æ–≥–æ —Ö–µ—à–∏—Ä–æ–≤–∞–Ω–Ω–æ–≥–æ –∑–Ω–∞—á–µ–Ω–∏—è
     * @param account –ê–¥—Ä–µ—Å –∞–∫–∫–∞—É–Ω—Ç–∞
     * @return True, –µ—Å–ª–∏ —Ö–µ—à–∏—Ä–æ–≤–∞–Ω–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –±—ã–ª–æ –¥–æ–±–∞–≤–ª–µ–Ω–æ, –∏–Ω–∞—á–µ - false
     */
    function hasHash(address account) public view returns (bool) {
        if (_hashes[account] != bytes32(0)) {
            return true;
        }

        return false;
    }

    /**
     * @notice –ü–æ–ª—É—á–∞–µ—Ç —Ö–µ—à –∑–Ω–∞—á–µ–Ω–∏–µ
     * @param account –ê–¥—Ä–µ—Å –∞–∫–∫–∞—É–Ω—Ç–∞
     * @param value –°–µ–∫—Ä–µ—Ç–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ, –∫–æ—Ç–æ—Ä–æ–µ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ –∑–∞—Ö–µ—à–∏—Ä–æ–≤–∞—Ç—å
     */
    function getHash(address account, string memory value) public pure returns (bytes32) {
        return keccak256(abi.encodePacked(account, value));
    }
}
