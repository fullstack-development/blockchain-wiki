```solidity
}
```

## Liquidation

### –í–æ–∑–º–æ–∂–Ω–æ—Å—Ç—å –ª–∏–∫–≤–∏–¥–∞—Ü–∏–∏ –∑–∞–π–º–∞

```solidity
function getHypotheticalAccountLiquidityInternal(
    address account,
    CToken cTokenModify,
    uint redeemTokens,
    uint borrowAmount
) internal view returns (Error, uint, uint) {
    /// –°—é–¥–∞ –∑–∞–ø–∏—Å—ã–≤–∞—é—Ç—Å—è –ø—Ä–æ–º–µ–∂—É—Ç–æ—á–Ω—ã–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –∏ –≤—ã—á–∏—Å–ª–µ–Ω–∏—è
    AccountLiquidityLocalVars memory vars;

    CToken[] memory assets = accountAssets[account];
    /// –û—Ä–≥–∞–Ω–∏–∑—É–µ–º —Ü–∏–∫–ª –¥–ª—è –ø–µ—Ä–µ–±–æ—Ä–∞ –≤—Å–µ—Ö –≤–ª–æ–∂–µ–Ω–Ω—ã—Ö –∞–∫—Ç–∏–≤–æ–≤ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è —Å –∞–¥—Ä–µ—Å–æ–º account
    for (uint i = 0; i < assets.length; i++) {
        CToken asset = assets[i];

        /// –ü–æ–¥–≥—Ä—É–∂–∞–µ–º –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—é –ø–æ –∫–∞–∂–¥–æ–º—É –∞–∫—Ç–∏–≤—É
        (oErr, vars.cTokenBalance, vars.borrowBalance, vars.exchangeRateMantissa) = asset.getAccountSnapshot(account);
        if (oErr != 0) {
            return (Error.SNAPSHOT_ERROR, 0, 0);
        }

        /// –î–æ–±–∞–≤–ª—è–µ–º —Ç–æ—á–Ω–æ—Å—Ç—å —Å–æ–≥–ª–∞—Å–Ω–æ –∫–æ–Ω—Ç—Ä–∞–∫—Ç—É "./ExponentialNoError.sol"
        vars.collateralFactor = Exp({mantissa: markets[address(asset)].collateralFactorMantissa});
        vars.exchangeRate = Exp({mantissa: vars.exchangeRateMantissa});

        /// –ü–æ–ª—É—á–∞–µ–º –ø—Ä–∞–π—Å –∏–∑ –æ—Ä–∞–∫—É–ª–∞ –¥–ª—è –±–∞–∑–æ–≤–æ–≥–æ –∞–∫—Ç–∏–≤–∞
        vars.oraclePriceMantissa = oracle.getUnderlyingPrice(asset);
        if (vars.oraclePriceMantissa == 0) {
            return (Error.PRICE_ERROR, 0, 0);
        }
        vars.oraclePrice = Exp({mantissa: vars.oraclePriceMantissa});

        /// –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º —Å—Ç–æ–∏–º–æ—Å—Ç—å –æ–¥–Ω–æ–≥–æ cToken
        vars.tokensToDenom = mul_(mul_(vars.collateralFactor, vars.exchangeRate), vars.oraclePrice);

        /// sumCollateral += tokensToDenom(—Å—Ç–æ–∏–º–æ—Å—Ç—å –æ–¥–Ω–æ cToken) * cTokenBalance(–∫–æ–ª–∏—á–µ—Å—Ç–≤–æ cToken —É –∞–∫–∫–∞—É–Ω—Ç–∞)
        vars.sumCollateral = mul_ScalarTruncateAddUInt(vars.tokensToDenom, vars.cTokenBalance, vars.sumCollateral);

        /// sumBorrowPlusEffects += oraclePrice(—Å—Ç–æ–∏–º–æ—Å—Ç—å –∞–∫—Ç–∏–≤–∞ —Å–æ–≥–ª–∞—Å–Ω–æ –æ—Ä–∞–∫—É–ª—É) * borrowBalance(–±–∞–ª–∞–Ω—Å –∑–∞–µ–º–Ω—ã—Ö —Å—Ä–µ–¥—Å—Ç–≤)
        vars.sumBorrowPlusEffects = mul_ScalarTruncateAddUInt(vars.oraclePrice, vars.borrowBalance, vars.sumBorrowPlusEffects);

        /// –î–ª—è –∞–∫—Ç–∏–≤–∞ –∫–æ—Ç–æ—Ä—ã–π –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å –±—Ä–∞–ª –≤ –∑–∞–π–º –∏ –ø–æ–∑–∏—Ü–∏—è –∫–æ—Ç–æ—Ä–æ–≥–æ –±—É–¥–µ—Ç –º–µ–Ω—è—Ç—å—Å—è
        if (asset == cTokenModify) {
            // sumBorrowPlusEffects += tokensToDenom(–ø—Ä–∞–π—Å –æ–¥–Ω–æ–≥–æ cToken) * redeemTokens(–∫–æ–ª–∏—á–µ—Å—Ç–≤–æ cToken –¥–ª—è –∑–∞–∫—Ä—ã—Ç–∏—è –ø–æ–∑–∏—Ü–∏–∏)
            vars.sumBorrowPlusEffects = mul_ScalarTruncateAddUInt(vars.tokensToDenom, redeemTokens, vars.sumBorrowPlusEffects);

            // sumBorrowPlusEffects += oraclePrice(—Å—Ç–æ–∏–º–æ—Å—Ç—å –∞–∫—Ç–∏–≤–∞) * borrowAmount (—Å—É–º–º–∞ —Å—Ä–µ–¥—Å—Ç–≤ –≤ –∑–∞–π–º–µ)
            vars.sumBorrowPlusEffects = mul_ScalarTruncateAddUInt(vars.oraclePrice, borrowAmount, vars.sumBorrowPlusEffects);
        }
    }

    /// –ó–∞—â–∏—Ç–∞ –æ—Ç –ø–µ—Ä–µ–ø–æ–ª–Ω–µ–Ω–∏—è
    if (vars.sumCollateral > vars.sumBorrowPlusEffects) {
        return (Error.NO_ERROR, vars.sumCollateral - vars.sumBorrowPlusEffects, 0);
    } else {
        return (Error.NO_ERROR, 0, vars.sumBorrowPlusEffects - vars.sumCollateral);
    }
}
```

    ```solidity
    function liquidateCalculateSeizeTokens(address cTokenBorrowed, address cTokenCollateral, uint actualRepayAmount) override external view returns (uint, uint) {
        /// –ü–æ–ª—É—á–µ–Ω–∏–µ —Å—Ç–æ–∏–º–æ—Å—Ç–∏ –∑–∞–ª–æ–≥–æ–≤—ã—Ö –∏ –∑–∞–µ–º–Ω—ã—Ö —Å—Ä–µ–¥—Å—Ç–≤ –∏–∑ –æ—Ä–∞–∫—É–ª–∞

        uint priceCollateralMantissa = oracle.getUnderlyingPrice(CToken(cTokenCollateral));
        if (priceBorrowedMantissa == 0 || priceCollateralMantissa == 0) {
            return (uint(Error.PRICE_ERROR), 0);
        }

        /// –ü–æ–ª—É—á–µ–Ω–∏–µ exchangeRate –¥–ª—è –æ–±–º–µ–Ω–∞ CToken

        uint seizeTokens;
        Exp memory numerator;
        Exp memory denominator;
        Exp memory ratio;

        /// –†–∞—Å—á–µ—Ç –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –∑–∞–ª–æ–≥–æ–≤–æ–≥–æ —Ç–æ–∫–µ–Ω–∞, –∫–æ—Ç–æ—Ä–æ–µ –±—É–¥–µ—Ç –ª–∏–∫–≤–∏–¥–∏—Ä–æ–≤–∞–Ω–æ

        denominator = mul_(Exp({mantissa: priceCollateralMantissa}), Exp({mantissa: exchangeRateMantissa}));

        seizeTokens = mul_ScalarTruncate(ratio, actualRepayAmount);

        return (uint(Error.NO_ERROR), seizeTokens);
    ```

    ```solidity
    function seizeInternal(address seizerToken, address liquidator, address borrower, uint seizeTokens) internal {
        /// –ü—Ä–æ–≤–µ—Ä—è–µ—Ç—Å—è –≤–æ–∑–º–æ–∂–Ω–æ—Å—Ç—å –≤—ã–∑–æ–≤–∞ —Ñ—É–Ω–∫—Ü–∏–∏ –Ω–∞ –∫–æ–Ω—Ç—Ä–∞–∫—Ç–µ Comptroller

        if (borrower == liquidator) {
            revert LiquidateSeizeLiquidatorIsBorrower();

        /*
         * –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º –Ω–æ–≤—ã–µ –±–∞–ª–∞–Ω—Å—ã –¥–ª—è –∑–∞–µ–º—â–∏–∫–∞ –∏ –ª–∏–∫–≤–∏–¥–∞—Ç–æ—Ä–∞

        uint liquidatorSeizeTokens = seizeTokens - protocolSeizeTokens;

        uint totalReservesNew = totalReserves + protocolSeizeAmount;

        /// –ù–µ–ø–æ—Å—Ä–µ–¥—Å—Ç–≤–µ–Ω–Ω–æ –æ–±–Ω–æ–≤–ª—è–µ–º –∑–Ω–∞—á–µ–Ω–∏—è –±–∞–ª–∞–Ω—Å–æ–≤ cToken –¥–ª—è –∑–∞–µ–º—â–∏–∫–∞ –∏ –ª–∏–∫–≤–∏–¥–∞—Ç–æ—Ä–∞

        totalSupply = totalSupply - protocolSeizeTokens;

        accountTokens[liquidator] = accountTokens[liquidator] + liquidatorSeizeTokens;

        emit Transfer(borrower, liquidator, liquidatorSeizeTokens);
        emit Transfer(borrower, address(this), protocolSeizeTokens);
        emit ReservesAdded(address(this), protocolSeizeAmount, totalReservesNew);
    }
    ```
